I decided on the 220 resistor for the Red and Greed LEDs because I have been consistently using them in class and in assignments.
To see work for how I mathmatically chose it, you can find the values plugged into ohms law in the A1 assignment where I found you need at least 160 ohms of resistance, so I chose the next closest higher number I have in resistors, 220.
10k Ohms for the button helps us to use less power. The button would work with less but 10k is a nice convention.

Code!

// int variable to represent pin 2 that will connect to button
const int buttonPin = 2;
// int variable to represent pin 13 that will connect to red led
const int ledPinRed = 13;
// int variable to represent pin 11 that will connect to green led
const int ledPinGreen = 11;
// int variable to represent pin 9 that will connect to blue led
const int ledPinBlue = 9;

// the setup function runs once when you press reset or power the board
void setup() {
// initialize digital pin ledPin as an output.
pinMode(ledPinRed, OUTPUT);
// initialize digital pin ledPinGreen as an output.
pinMode(ledPinGreen, OUTPUT);
// initialize digital pin ledPinBlue as an output.
pinMode(ledPinBlue, OUTPUT);
// initialize digital pin buttonPin as an input.
pinMode(buttonPin, INPUT);
}

// the loop function runs over and over again forever
void loop() {
// checks to see if the button in being pressed, if high it is.
if (digitalRead(buttonPin) == HIGH) {
// Turns on red led.
digitalWrite(ledPinRed, HIGH);
// Turns on blue led.
digitalWrite(ledPinBlue, HIGH);
//turns on green led.
digitalWrite(ledPinGreen, HIGH);

// for loop to incrimentally increase the brightness of the blue and green leds
for (int i = 0; i <= 255; i++) {
// sets green led to brightness of 'i'
analogWrite(ledPinGreen, i);
// sets blue led to brightness of 'i'
analogWrite(ledPinBlue, i);
// waits 0.05 seconds before returning to top of for loop
delay(5);
}

// for loop to incrimentally decrease the brightness of the blue and green leds
for (int i = 255; i >= 0; i--) {
// sets green led to brightness of 'i'
analogWrite(ledPinGreen, i);
// sets blue led to brightness of 'i'
analogWrite(ledPinBlue, i);
// waits 0.05 sec
}
// if buttonpin reads "low" aka button is not pressed.
} else {
// Red led turns off.
digitalWrite(ledPinRed, LOW);
// Green led turns off.
digitalWrite(ledPinGreen, LOW);
// blue led turns off
digitalWrite(ledPinBlue, LOW);
}
}
